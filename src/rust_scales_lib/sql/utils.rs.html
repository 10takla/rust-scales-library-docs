use crate::model::dto::product::ProductDB;
use diesel::query_builder::IntoUpdateTarget;
use diesel::{debug_query, dsl::delete, query_builder::QueryFragment, sqlite::Sqlite};
use serde::Serialize;

pub fn delete_to_string<T: QueryFragment<Sqlite>>(table: T) -> String {
    semicolon_format(query_without_binds(&debug_query::<Sqlite, _>(&table).to_string()).into())
}

pub fn inserts_to_string<T: QueryFragment<Sqlite>>(queries: Vec<(T, Vec<String>)>) -> String {
    queries
        .into_iter()
        .map(|(query, values)| {
            insert_into_to_string(query, values).replace("INSERT INTO", "INSERT OR REPLACE INTO")
        })
        .collect::<Vec<_>>()
        .join("\n")
}

fn query_without_binds<'a>(query: &'a String) -> &'a str {
    query.split_once(" -- binds").unwrap().0
}

fn semicolon_format(query: String) -> String {
    format!("{query};")
}

fn insert_into_to_string(query: impl QueryFragment<Sqlite>, values: Vec<String>) -> String {
    let query = debug_query::<Sqlite, _>(&query).to_string();
    let query = query_without_binds(&query);

    let str_sql = values
        .into_iter()
        .fold(query.to_string(), |acc, value| acc.replacen("?", &value, 1));
    semicolon_format(str_sql)
}

pub mod native_sql {
    use crate::model::dto::shared::sqlite_types::{SQLiteMap, SQLiteValue};

    pub fn without<const L: usize, M: SQLiteMap<L>>(
        entries: impl AsRef<[M]>,
        table: &str,
    ) -> String {
        let entries = entries.as_ref();
        if entries.len() == 0 {
            return String::new();
        }
        format!(
            "{}\n{};",
            get_head_insert_into(&entries[0].get_map()[0], table),
            entries
                .into_iter()
                .map(|entry| wrap_expr(&entry.get_map()[1]))
                .collect::<Vec<_>>()
                .join(",\n")
        )
    }

    pub fn native_insert_into_entries<const L: usize, M: SQLiteMap<L>>(
        entries: impl AsRef<[M]>,
        table: &str,
    ) -> String {
        let entries = entries.as_ref();
        if entries.len() == 0 {
            return String::new();
        }
        format!(
            "{}\n{};",
            get_head_insert_into(&entries[0].get_keys(), table),
            entries
                .into_iter()
                .map(|entry| wrap_expr(&entry.get_values()))
                .collect::<Vec<_>>()
                .join(",\n")
        )
    }

    pub fn native_insert_into_select_entry<const L: usize, M: SQLiteMap<L>>(
        value: M,
        table: &str,
    ) -> String {
        let [keys, values] = [value.get_keys(), value.get_values()];
        format!(
            "INSERT INTO {table} ({})\nSELECT {}\nWHERE NOT EXISTS (SELECT {} FROM {table} WHERE {} = {});",
            keys.join(", "),
            values.join(", "),
            values[0], keys[0], values[0]
        )
    }

    pub fn native_insert_into<const L: usize>(entry: impl SQLiteMap<L>, table: &str) -> String {
        format!(
            "{} {};",
            get_head_insert_into(dbg!(&entry.get_keys()), table),
            wrap_expr(&entry.get_values())
        )
    }

    pub fn insert_by_keys_values(entries: Vec<[impl AsRef<[String]>; 2]>, table: &str) -> String {
        if entries.len() == 0 {
            return String::new();
        }
        format!(
            "{}\n{};",
            get_head_insert_into(&entries[0][0].as_ref(), table),
            entries
                .into_iter()
                .map(|entry| wrap_expr(&entry[1].as_ref()))
                .collect::<Vec<_>>()
                .join(",\n")
        )
    }

    fn get_head_insert_into(keys: &[String], table: &str) -> String {
        format!(
            "INSERT OR REPLACE INTO {table} {} VALUES",
            wrap_expr(
                &keys
                    .into_iter()
                    .map(|key| format!("`{key}`"))
                    .collect::<Vec<_>>()
            )
        )
    }
    fn wrap_expr(vec: &[String]) -> String {
        format!("({})", vec.join(", "))
    }
